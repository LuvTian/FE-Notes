PIE_IE678.js解决ie8支持部分css属性 ,还需要初始化一下，有时候需要给元素加上position:relative;z-index:0;属性才能失效，在css中引用PIE.htc文件一定要使用相对于html文件的路径，而不是相对于css文件的路径

$(function() {
	if (window.PIE) {
        $('img,.circle-border-btn-wrap,.circle-border-btn-wrap .circle-border-btn').each(function() {
            PIE.attach(this);
        });
    }
})
